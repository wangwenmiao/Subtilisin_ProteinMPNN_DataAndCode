--- 文件开始: step4_heat_nvt_373K.in ---
NVT heating from 300K to 373K with protein backbone restraints
&cntrl
  imin=0, irest=1, ntx=5,  ! MD from scratch, no velocities in input coords
  nstlim=30000,            ! 60 ps (30000 * 0.002 ps/step)
  dt=0.001,                ! 1.0 fs time step
  ntc=2, ntf=2,            ! SHAKE on for bonds involving hydrogen
  cut=10.0,
  ntb=1,                   ! Constant volume periodic boundary (NVT)
  ntpr=500,                ! Print energy info every 1.0 ps
  ntwx=500,                ! Write coordinates every 1.0 ps to trajectory
  ntwr=5000,               ! Write restart file every 10 ps
  ntt=3,                   ! Langevin thermostat
  gamma_ln=2.0,            ! Collision frequency (ps^-1)
  tempi=300.0,               ! Initial temperature for velocity generation
  temp0=373.0,             ! Target temperature (will be ramped)
  ig=1,                   ! Random seed for velocity generation
  ntr=1,                   ! Positional restraints on
  restraintmask=':1-1000 & @C,CA,N', ! Restrain protein backbone atoms (C, CA, N)
  restraint_wt=4.0,        ! Medium restraint force constant (kcal/mol/A^2)
  nmropt=1,                ! Activate NMR options for TEMP0 ramp
  ioutfm=1,                ! Write trajectory in NetCDF format
  ntxo=2,                  ! Write restart file in NetCDF format
/
&wt TYPE='TEMP0', istep1=0, istep2=25000,   ! Ramp temperature over 30 ps
  value1=300.0, value2=373.0 /                ! from 0.1 K to 300.0 K
&wt TYPE='END' /

Minimization of solvent and ions with protein heavy atoms restrained
&cntrl
  imin=1,           ! Perform energy minimization
  maxcyc=10000,      ! Maximum number of minimization cycles
  ncyc=1000,        ! First 1000 cycles steepest descent, then conjugate gradient
  ntmin=2,          ! Steepest descent then conjugate gradient
  cut=10.0,         ! Cutoff for non-bonded interactions (Angstroms)
  ntb=1,            ! Constant volume periodic boundary
  ntc=1,            ! No SHAKE for bonds
  ntf=1,            ! No SHAKE, calculate all forces
  ntpr=100,         ! Print energy information every 100 steps
  ntwr=1000,        ! Write restart file every 1000 steps (or at end)
  ntr=1,            ! Use positional restraints
  restraintmask=':1-1000 & !@H=', ! Restrain all heavy atoms of the protein
  restraint_wt=10.0,  ! Restraint force constant (kcal/mol/A^2) - Strong restraint
  ntxo=2,           ! Write restart file in NetCDF format
/
